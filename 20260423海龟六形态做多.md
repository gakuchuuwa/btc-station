strategy('Turtle 6-Pattern 4H 只做多',
         overlay=true, 
         initial_capital=10000, 
         commission_type=strategy.commission.percent, 
         commission_value=0.06, 
         slippage=500,
         default_qty_value=1, 
         default_qty_type=strategy.percent_of_equity, 
         margin_long=1, 
         pyramiding=2,
         max_labels_count=500,
         max_lines_count=500)

// ==================== MA计算函数 ====================
ma(source, length, type) =>
    switch type
        'RMA' => ta.rma(source, length)
        'SMA' => ta.sma(source, length)
        'EMA' => ta.ema(source, length)
        'WMA' => ta.wma(source, length)
        'VWMA' => ta.vwma(source, length)

// ==================== 输入参数 ====================
// 形态开关
enable_p1 = input.bool(true, '形态1（突破做多）', group='形态开关', tooltip='形态1：突破S1或S2上轨做多')
enable_p2 = input.bool(true, '形态2（MACD回调做多）', group='形态开关', tooltip='形态2：跌破S2下轨后等待MACD金叉做多')
enable_p3 = input.bool(true, '形态3（MACD回调做多）', group='形态开关', tooltip='形态3：跌破S2下轨后等待MACD金叉做多')
enable_p4 = input.bool(true, '形态4（MACD回调做多）', group='形态开关', tooltip='形态4：跌破S2下轨后等待MACD金叉做多')
enable_p5 = input.bool(true, '形态5（突破做多）', group='形态开关', tooltip='形态5：突破S1或S2上轨做多')
enable_p6 = input.bool(true, '形态6（突破做多）', group='形态开关', tooltip='形态6：突破S1或S2上轨做多')


enable_p1_switch_exit = input.bool(true, '启用P1形态切换平多', group='形态切换平仓', tooltip='持多到达P1后，形态切换至其他（如P2/P3/P6）时平多')

// MACD
macd_fast = input.int(13, 'MACD快线周期', minval=1, group='MACD')
macd_slow = input.int(21, 'MACD慢线周期', minval=1, group='MACD')
macd_signal = input.int(8, 'MACD信号线周期', minval=1, group='MACD')

enable_obv_filter = input.bool(true, '启用OBV过滤器', group='OBV过滤器', tooltip='OBV振荡器>0才允许做多。基于LazyBear的OBV Oscillator。')
obv_length = input.int(20, 'OBV振荡器EMA周期', minval=1, group='OBV过滤器')

// 双系统配置
use_dual_system = input.bool(true, '启用双系统突破', group='双系统配置')
use_s1_filter = input.bool(true, '系统1失败过滤', group='双系统配置', tooltip='系统1亏损后禁用，直到系统2有一笔盈利才重新启用。')

// 系统1（短周期）
system1_period = input.int(55, 'S1周期（短周期）', minval=12, step=6, group='系统1')

// 系统2（长周期）
system2_period = input.int(144, 'S2周期（长周期）', minval=24, step=6, group='系统2')

// 资金管理
base_risk_percent = input.float(1.0, '全局基础风险比例(%)', minval=0.1, step=0.1, group='资金管理 - 全局', tooltip='每笔交易占总资金的基础风险百分比。')
use_real_capital = input.bool(true, '实盘模式（固定本金）', group='资金管理 - 全局', tooltip='开=实盘模式（使用固定本金）\n关=回测模式（复利计算）')
real_capital = input.float(10000, '实盘本金（USDT）', minval=100, group='资金管理 - 全局')
enable_max_qty = input.bool(true, '启用最大开仓量上限', group='资金管理 - 全局', tooltip='交易BTC时建议开启，交易其他品种（合约规格不同）时建议关闭')
max_qty = input.float(10.0, '最大开仓量上限（BTC）', minval=0.001, step=1.0, group='资金管理 - 全局', tooltip='单次开仓最大BTC手数上限，防止极端行情下浮点计算暴仓')
// OKX BTC 合约精度（如合约规格变更可在面板调整）
float qty_step = 0.001
float min_qty  = 0.001

// 形态仓位倍数
mult_p1 = input.float(4.0, 'P1仓位倍数（主力）', minval=0.0, step=1, group='形态仓位倍数')
mult_p2 = input.float(2.0, 'P2仓位倍数（辅助）', minval=0.0, step=1, group='形态仓位倍数')
mult_p3 = input.float(1.0, 'P3仓位倍数（减仓）', minval=0.0, step=1, group='形态仓位倍数')
mult_p4 = input.float(3.0, 'P4仓位倍数（防御）', minval=0.0, step=1, group='形态仓位倍数')
mult_p5 = input.float(2.0, 'P5仓位倍数（辅助）', minval=0.0, step=1, group='形态仓位倍数')
mult_p6 = input.float(1.0, 'P6仓位倍数（基础）', minval=0.0, step=1, group='形态仓位倍数')

// 价格止损
enable_realtime_stop = input.bool(true, '启用价格止损', group='价格止损')
realtime_stop_percent = input.float(4.0, '止损比例(%)', minval=1.0, maxval=10.0, step=1, group='价格止损')

// ATR止损
atr_period = system1_period
atr_mult_init = input.float(2.0, 'ATR初始止损倍数', minval=0.1, step=1, group='ATR止损')
atr_mult_long_trail = input.float(12.0, 'ATR追踪止损倍数（多头）', minval=0.1, step=1, group='ATR止损')

// 追踪止损激活
enable_trailing_stop = input.bool(true, '启用追踪止损', group='追踪止损激活')
profit_atr_mult_long = input.float(5.0, '追踪止损激活ATR倍数（多头）', minval=0.5, step=1, group='追踪止损激活')
use_breakeven = input.bool(true, '激活追踪时保本止损', group='追踪止损激活')

pyramid_enable = input.bool(true, '启用加仓（金字塔）', group='加仓设置')
pyramid_max_count_long  = input.int(1, '最大加仓次数（多头）', minval=0, maxval=1, step=1, group='加仓设置')
pyramid_mult_u2 = input.float(0.5, '加仓数量倍数', minval=0.1, maxval=5.0, step=0.1, group='加仓设置')

// 通道1
ma1_length = input.int(144, 'MA1周期', minval=1, step=24, group='通道1')
ma1_type = input.string('RMA', 'MA1类型', options=['SMA', 'EMA', 'RMA', 'WMA', 'VWMA'], group='通道1')
ma1_color = input(color.orange, '上轨颜色', inline='c1', group='通道1')
ma2_color = input(color.aqua, '下轨颜色', inline='c1', group='通道1')

// 通道2
ma3_length = input.int(576, 'MA2周期', minval=1, step=24, group='通道2')
ma3_type = input.string('RMA', 'MA2类型', options=['SMA', 'EMA', 'RMA', 'WMA', 'VWMA'], group='通道2')
ma3_color = input(color.yellow, '上轨颜色', inline='c2', group='通道2')
ma4_color = input(color.blue, '下轨颜色', inline='c2', group='通道2')

// ==================== 指标计算 ====================
float atr_value = ta.atr(atr_period)

// SSL通道计算
float ma1 = ma(high, ma1_length, ma1_type)
float ma2 = ma(low, ma1_length, ma1_type)
float ma3 = ma(high, ma3_length, ma3_type)
float ma4 = ma(low, ma3_length, ma3_type)

// MACD计算
[macdLine, signalLine, macdHist] = ta.macd(close, macd_fast, macd_slow, macd_signal)
bool macd_golden_cross = ta.crossover(macdLine, signalLine)

// OBV振荡器计算（LazyBear OBV Oscillator）
// OBV = 累计量能（上涨+volume，下跌-volume，平盘0）
float obv_value = ta.cum(ta.change(close) > 0 ? volume : ta.change(close) < 0 ? -volume : 0)
float obv_osc = obv_value - ta.ema(obv_value, obv_length)
bool obv_allow_long  = not enable_obv_filter or obv_osc > 0

var float Hlv1 = na
Hlv1 := close > ma1 ? 1 : close < ma2 ? -1 : Hlv1[1]
float sslUp1 = Hlv1 < 0 ? ma2 : ma1
float sslDown1 = Hlv1 < 0 ? ma1 : ma2
color color1 = Hlv1 == 1 ? ma1_color : ma2_color

var float Hlv2 = na
Hlv2 := close > ma3 ? 1 : close < ma4 ? -1 : Hlv2[1]
float sslUp2 = Hlv2 < 0 ? ma4 : ma3
float sslDown2 = Hlv2 < 0 ? ma3 : ma4
color color2 = Hlv2 == 1 ? ma3_color : ma4_color

plot(sslUp1, 'UP1', linewidth=1, color=color1)
plot(sslDown1, 'DOWN1', linewidth=1, color=color1)
plot(sslUp2, 'UP2', linewidth=1, color=color2)
plot(sslDown2, 'DOWN2', linewidth=1, color=color2)

// ==================== 双系统唐奇安通道 ====================
// 系统1通道（短周期）
float s1_high_long = ta.highest(high, system1_period)[1]
float s1_low_short = ta.lowest(low, system1_period)[1]

// 系统2通道（长周期）
float s2_high_long = ta.highest(high, system2_period)[1]
float s2_low_short = ta.lowest(low, system2_period)[1]


// ==================== 形态识别 ====================
bool c1_gt_c2 = ma1 > ma3 and ma2 > ma4
bool c1_lt_c2 = ma1 < ma3 and ma2 < ma4

var int current_pattern = 1

if Hlv1 == 1 and Hlv2 == 1 and c1_gt_c2
    current_pattern := 1
else if Hlv1 == -1 and Hlv2 == 1 and c1_gt_c2
    current_pattern := 2
else if Hlv1 == -1 and Hlv2 == -1 and c1_gt_c2
    current_pattern := 3
else if Hlv1 == -1 and Hlv2 == -1 and c1_lt_c2
    current_pattern := 4
else if Hlv1 == 1 and Hlv2 == -1 and c1_lt_c2
    current_pattern := 5
else if Hlv1 == 1 and Hlv2 == 1 and c1_lt_c2
    current_pattern := 6

// ==================== 状态变量 ====================
var float cycle_start_equity = strategy.initial_capital
var float long_stop_level = na
var bool long_trail_active = false
var bool long_trail_just_activated = false
var int entry_pattern_num = 0
var float entry_atr = na
var bool has_reached_p1 = false  // 持多时是否到达过P1
var float entry_risk_percent = na
var float entry_size_multiplier = na
var string display_system = na

// 双系统互锁状态变量
var bool s1_last_failed = false
var string entry_system = na

// MACD就绪标记
var bool p2_dip_ready = false
var bool p3_dip_ready = false
var bool p4_dip_ready = false

// 加仓状态变量
var int pyramid_count_long = 0   // 多头已加仓次数

// ==================== 平仓检测 ====================
bool position_closed = strategy.position_size == 0 and strategy.position_size[1] != 0

// 保存entry_system用于互锁判断
string saved_entry_system = entry_system

if position_closed
    // 检测系统1/2盈亏并更新互锁状态
    if strategy.closedtrades > 0
        int i = strategy.closedtrades - 1
        float total_profit = strategy.closedtrades.profit(i)
        // 遍历所有共享同一 exit_time 的订单，累加总盈亏（兼容任意 pyramiding 值）
        int j = i - 1
        while j >= 0 and strategy.closedtrades.exit_time(j) == strategy.closedtrades.exit_time(i)
            total_profit += strategy.closedtrades.profit(j)
            j -= 1
        bool last_trade_profit = total_profit > 0
        if saved_entry_system == 'S1' and use_s1_filter
            s1_last_failed := not last_trade_profit
        if saved_entry_system == 'S2' and last_trade_profit
            s1_last_failed := false

    long_stop_level := na
    long_trail_active := false
    long_trail_just_activated := false
    entry_pattern_num := 0
    entry_atr := na
    has_reached_p1 := false
    p2_dip_ready := false
    p3_dip_ready := false
    p4_dip_ready := false
    display_system := na
    entry_system := na
    entry_risk_percent := na
    entry_size_multiplier := na
    pyramid_count_long := 0

if strategy.position_size == 0 and strategy.position_size[1] == 0
    cycle_start_equity := strategy.equity

// ==================== 动态仓位计算 (带OKX精度控制) ====================
calculate_position_size(entry_price, atr_value, atr_multiplier, risk_percent, size_multiplier) =>
    float account_equity = use_real_capital ? real_capital : strategy.equity
    float actual_risk_percent = risk_percent * size_multiplier
    float risk_amount = account_equity * actual_risk_percent / 100
    float atr_stop_distance = atr_value * atr_multiplier
    float safe_stop_distance = math.max(atr_stop_distance, 0.01)
    float raw_qty = risk_amount / safe_stop_distance
    // 【优化】加 1e-9 容差对抗IEEE-754浮点误差，math.round(...,6)抹除微尘保证JSON干净
    float rounded_qty = math.round(math.floor(raw_qty / qty_step + 1e-9) * qty_step, 6)
    float capped_qty  = enable_max_qty ? math.min(rounded_qty, max_qty) : rounded_qty
    // 低于最小下单量则返回0，调用方需检查
    float final_qty = capped_qty < min_qty ? 0.0 : capped_qty
    final_qty

// ==================== 双系统突破信号 ====================
// 系统1信号（短周期）
bool s1_breakout_long = close > s1_high_long and close[1] <= s1_high_long[1]

// 系统2信号（长周期）
bool s2_breakout_long = close > s2_high_long and close[1] <= s2_high_long[1]

// ==================== MACD就绪状态更新 ====================
// P2/P3/P4: 跌破S2下轨就绪（做多回调接力）
if current_pattern != 2 and current_pattern != 3 and current_pattern != 4
    p2_dip_ready := false
    p3_dip_ready := false
    p4_dip_ready := false
if (current_pattern == 2 and enable_p2) or (current_pattern == 3 and enable_p3) or (current_pattern == 4 and enable_p4)
    if close < s2_low_short
        p2_dip_ready := true
        p3_dip_ready := true
        p4_dip_ready := true

// ==================== 入场信号生成 ====================
string signal_long = na

// S1信号（带失败过滤）
bool s1_long_ok = use_dual_system and s1_breakout_long and strategy.position_size == 0
if use_s1_filter and s1_last_failed
    s1_long_ok := false

// P1/P5/P6: 突破做多组（S1或S2）
bool is_breakout_pattern = (current_pattern == 1 and enable_p1) or (current_pattern == 5 and enable_p5) or (current_pattern == 6 and enable_p6)
if is_breakout_pattern and strategy.position_size == 0
    if s1_long_ok
        signal_long := 'S1'
    else if s2_breakout_long
        signal_long := 'S2'

// P2/P3/P4: 突破下限MACD回调做多组
bool is_dip_pattern = (current_pattern == 2 and enable_p2) or (current_pattern == 3 and enable_p3) or (current_pattern == 4 and enable_p4)
bool dip_ready = (current_pattern == 2 and p2_dip_ready) or (current_pattern == 3 and p3_dip_ready) or (current_pattern == 4 and p4_dip_ready)
if is_dip_pattern and dip_ready and macd_golden_cross and strategy.position_size == 0
    if na(signal_long)
        signal_long := 'MACD'

// ==================== OBV过滤器执行 ====================
// OBV振荡器 > 0 才允许做多
if not obv_allow_long
    signal_long := na

// ==================== 出场条件 ====================
// P1形态切换平多：持多到达P1后，形态再切换则平多
if strategy.position_size > 0 and current_pattern == 1
    has_reached_p1 := true

bool exit_long_pattern = enable_p1_switch_exit and strategy.position_size > 0 and has_reached_p1 and current_pattern != 1

// ==================== 风险参数选择 ====================
get_risk_params(pattern) =>
    float m = 1.0
    if pattern == 1
        m := mult_p1
    else if pattern == 2
        m := mult_p2
    else if pattern == 3
        m := mult_p3
    else if pattern == 4
        m := mult_p4
    else if pattern == 5
        m := mult_p5
    else if pattern == 6
        m := mult_p6
    [base_risk_percent, m]

// ==================== 做多入场 ====================
if not na(signal_long)
    cycle_start_equity := strategy.equity
    entry_pattern_num := current_pattern
    entry_atr := atr_value
    has_reached_p1 := (current_pattern == 1)  // 修复：进场即P1则立刻激活，防止形态秒切导致退出逻辑失效
    p2_dip_ready := false
    p3_dip_ready := false
    p4_dip_ready := false

    // 【修复】底层授权与 UI 展示分离，保护 S1 失败过滤器不失效
    entry_system := signal_long     // 永远真实记录是 S1 还是 S2
    display_system := signal_long


    [current_risk, current_mult] = get_risk_params(current_pattern)
    entry_risk_percent := current_risk
    entry_size_multiplier := current_mult

    float qty = calculate_position_size(close, atr_value, atr_mult_init, current_risk, current_mult)

    if qty > 0
        string entry_id = 'Pattern' + str.tostring(current_pattern) + 'Long'
        string comment = display_system + ' Long-P' + str.tostring(current_pattern)
        strategy.entry(entry_id, strategy.long, qty=qty, comment=comment)
        pyramid_count_long := 0

// ==================== 出场执行 ====================
if exit_long_pattern
    strategy.close_all(comment='P1 Pattern Switch Exit Long')

// ==================== 多头ATR止损 ====================
if strategy.position_size > 0
    if strategy.position_size[1] <= 0
        long_stop_level := strategy.position_avg_price - entry_atr * atr_mult_init
        long_trail_active := false
        long_trail_just_activated := false
    else
        float current_profit = high - strategy.position_avg_price
        if enable_trailing_stop
            if not long_trail_active and current_profit >= atr_value * profit_atr_mult_long
                long_trail_active := true
                long_trail_just_activated := true
                float calculated_trail_long = high - atr_value * atr_mult_long_trail
                long_stop_level := use_breakeven ? math.max(strategy.position_avg_price, calculated_trail_long) : calculated_trail_long
            if long_trail_active and not long_trail_just_activated
                float new_stop = close - atr_value * atr_mult_long_trail
                long_stop_level := math.max(long_stop_level, new_stop)

    if not na(long_stop_level)
        if not long_trail_active
            if close < long_stop_level
                strategy.close_all(comment='Initial SL')
        else if not long_trail_just_activated
            if close < long_stop_level
                strategy.close_all(comment='Trailing SL')


// ==================== 多头加仓（追踪止损激活时加仓，支持0/1/2次） ====================
if strategy.position_size > 0 and pyramid_enable and pyramid_max_count_long > 0 and pyramid_count_long < pyramid_max_count_long
    if long_trail_just_activated
        pyramid_count_long := pyramid_count_long + 1
        float base_qty = calculate_position_size(close, atr_value, atr_mult_init, entry_risk_percent, entry_size_multiplier)
        float add_qty = math.round(math.floor(base_qty * pyramid_mult_u2 / qty_step + 1e-9) * qty_step, 6)
        if add_qty >= min_qty
            string add_comment = 'Scale-in Long ' + str.tostring(pyramid_count_long)
            strategy.entry('Scale-in Long' + str.tostring(pyramid_count_long), strategy.long, qty=add_qty, comment=add_comment)

// 多头加仓后下一根K线更新保本止损到新均价
if strategy.position_size > 0 and pyramid_count_long > 0 and long_trail_active and not long_trail_just_activated
    if use_breakeven
        long_stop_level := math.max(long_stop_level, strategy.position_avg_price)

// 重置追踪止损激活标志
if long_trail_just_activated
    long_trail_just_activated := false

if enable_realtime_stop
    if strategy.position_size > 0
        float realtime_stop_price_long = strategy.position_avg_price * (1 - realtime_stop_percent / 100)
        strategy.exit('Price SL Long', stop=realtime_stop_price_long, comment='Price SL')

var float realtime_stop_display = na
if strategy.position_size > 0
    realtime_stop_display := enable_realtime_stop ? strategy.position_avg_price * (1 - realtime_stop_percent / 100) : na
else
    realtime_stop_display := na

plot(strategy.position_avg_price, '入场均价', color=color.new(color.gray, 44), linewidth=1, style=plot.style_linebr)
plot(strategy.position_size > 0 ? long_stop_level : na, '多头ATR止损', color=color.new(color.fuchsia, 44), linewidth=1, style=plot.style_linebr)
plot(realtime_stop_display, '实时止损线', color=color.new(color.maroon, 44), linewidth=1, style=plot.style_linebr)

// 双系统通道线
plot(use_dual_system ? s1_high_long : na, 'S1多头上轨', color=color.new(color.lime, 44), linewidth=1)
plot(use_dual_system ? s1_low_short : na, 'S1下轨', color=color.new(color.red, 44), linewidth=1)
plot(s2_high_long, 'S2上轨', color=color.new(color.lime, 11), linewidth=2)
plot(s2_low_short, 'S2下轨', color=color.new(color.red, 11), linewidth=2)




// 形态标签
bool pattern_changed = current_pattern != current_pattern[1]
plotshape(pattern_changed[1] and current_pattern[1] == 1, style=shape.labeldown, location=location.abovebar, color=color.green,  text='1', textcolor=color.white, size=size.small)
