Hlv2 := close > ma3 ? 1 : close < ma4 ? -1 : Hlv2[1]

float sslDown2 = Hlv2 < 0 ? ma3 : ma4

    current_pattern := 5

var bool bull_market_active = true    // 初始为牛市

var bool p6_short_locked = false      // P6做空止损后锁定，跌回P4才解锁
var string regime_test_type = na

if enable_regime_filter
    // 【优化核心】熊→牛: 只要通道变成多头排列 (c1_gt_c2)，立刻切回牛市
    if c1_gt_c2

var bool long_trail_active = false

var int entry_pattern_num = 0
var float entry_atr = na
var bool has_reached_p1 = false  // 牛市持多时是否到达过P1

var float entry_size_multiplier = na

string saved_entry_system      = entry_system
int    saved_entry_pattern_num = entry_pattern_num

if position_closed
    // 检测系统1/2盈亏并更新互锁状态
    if strategy.closedtrades > 0
        int i = strategy.closedtrades - 1
        float total_profit = strategy.closedtrades.profit(i)
        // 遍历所有共享同一 exit_time 的订单，累加总盈亏（兼容任意 pyramiding 值）
        int j = i - 1
        while j >= 0 and strategy.closedtrades.exit_time(j) == strategy.closedtrades.exit_time(i)
            total_profit += strategy.closedtrades.profit(j)
            j -= 1
        bool last_trade_profit = total_profit > 0
        if saved_entry_system == 'S1' and use_s1_filter
            s1_last_failed := not last_trade_profit
        if saved_entry_system == 'S2' and last_trade_profit
            s1_last_failed := false

        // 开关A：追踪P1多头损失，记录冷却期起始bar
        if enable_p1_cooldown and saved_entry_pattern_num == 1 and strategy.position_size[1] > 0 and not last_trade_profit
            p1_loss_bar := bar_index

        // P5/P6做空止损：锁定做空权限
        if bear_market_active and (entry_pattern_num == 5 or entry_pattern_num == 6) and strategy.position_size[1] < 0
            if not last_trade_profit
                p6_short_locked := true       // 锁定P5/P6做空，等跌回P4才解锁

        // 牛熊切换盈亏确认
        if enable_regime_filter
            if regime_test_type == 'P4_bear_test'
                // P4做空盈利 或 P4触发S2下轨突破（无论盈亏）→ 确认切熊
                if last_trade_profit or saved_entry_system == 'S2'
                    bull_market_active := false
                    bear_market_active := true

            regime_test_type := na

    short_stop_level := na
    short_trail_active := false

    entry_atr := na

    p3_dip_ready := false
    p5_breakout_ready := false
    p6_breakout_ready := false

    entry_risk_percent := na
    entry_size_multiplier := na
    entry_price_first := na
    pyramid_count_long := 0

    cycle_start_equity := strategy.equity

// ==================== 动态仓位计算 (带OKX精度控制) ====================
calculate_position_size(entry_price, atr_value, atr_multiplier, risk_percent, size_multiplier) =>
    float account_equity = use_real_capital ? real_capital : strategy.equity
    float actual_risk_percent = risk_percent * size_multiplier
    float risk_amount = account_equity * actual_risk_percent / 100
    float atr_stop_distance = atr_value * atr_multiplier
    float safe_stop_distance = math.max(atr_stop_distance, 0.01)
    float raw_qty = risk_amount / safe_stop_distance
    // 【优化】加 1e-9 容差对抗IEEE-754浮点误差，math.round(...,6)抹除微尘保证JSON干净
    float rounded_qty = math.round(math.floor(raw_qty / qty_step + 1e-9) * qty_step, 6)
    float capped_qty  = enable_max_qty ? math.min(rounded_qty, max_qty) : rounded_qty
    // 低于最小下单量则返回0，调用方需检查
    float final_qty = capped_qty < min_qty ? 0.0 : capped_qty
    final_qty

    p2_dip_ready := false

    if close < s2_low_short

        p3_dip_ready := true

    p6_breakout_ready := false

    if s2_breakout_long

// ==================== 风险参数选择 (适配9形态) ====================
get_risk_params(pattern, is_long) =>
    float m = 1.0
    if is_long
        if pattern == 1
            m := mult_p1_l
        else if pattern == 2
            m := mult_p2_l
        else if pattern == 3
            m := mult_p3_l
        else if pattern == 5
            m := mult_p5_l
        else if pattern == 6
            m := mult_p6_l
    else
        if pattern == 4
            m := mult_p4_s
        else if pattern == 5
            m := mult_p5_s
        else if pattern == 6
            m := mult_p6_s
    [base_risk_percent, m]

// ==================== 入场信号生成 (9独立形态精简版) ====================

string signal_short = na

bool s1_long_ok = use_dual_system and s1_breakout_long and strategy.position_size == 0
bool s1_short_ok = use_dual_system and s1_breakout_short and strategy.position_size == 0
if use_s1_filter and s1_last_failed
    s1_long_ok := false
    s1_short_ok := false

if enable_regime_filter
    // === 牛市 === (涵盖 P1, P2, P3, P5多, P6多, P4空试探)
    if bull_market_active
        // 1. 突破做多 (P1, P5, P6)
        if (current_pattern == 1 or current_pattern == 5 or current_pattern == 6) and strategy.position_size == 0
            if s1_long_ok
                signal_long := 'S1'
            else if s2_breakout_long
                signal_long := 'S2'

        // 2. MACD接力做多 (P2, P3)
        if (current_pattern == 2 and p2_dip_ready) or (current_pattern == 3 and p3_dip_ready)
            if macd_golden_cross and strategy.position_size == 0 and na(signal_long)
                signal_long := 'MACD'

        // 3. 试探做空 (P4)
        if current_pattern == 4 and strategy.position_size == 0
            if s1_short_ok
                signal_short := 'S1'
            else if s2_breakout_short
                signal_short := 'S2'

    // === 熊市 === (涵盖 P4空, P5空, P6空)
    else if bear_market_active
        // 5. 突破做空 (P4)
        if current_pattern == 4 and strategy.position_size == 0
            if s1_short_ok
                signal_short := 'S1'
            else if s2_breakout_short
                signal_short := 'S2'

        // 6. MACD接力做空 (P5, P6)
        if current_pattern == 5 and p5_breakout_ready and macd_death_cross and not p6_short_locked and strategy.position_size == 0
            if na(signal_short)
                signal_short := 'MACD'
        if current_pattern == 6 and p6_breakout_ready and macd_death_cross and not p6_short_locked and strategy.position_size == 0
            if na(signal_short)
                signal_short := 'MACD'

else
    // === 牛熊过滤器关闭：视为永远牛市 ===
    // 1. 突破做多 (P1, P5, P6)
    if (current_pattern == 1 or current_pattern == 5 or current_pattern == 6) and strategy.position_size == 0
        if s1_long_ok
            signal_long := 'S1'
        else if s2_breakout_long
            signal_long := 'S2'

    // 2. MACD接力做多 (P2, P3)
    if (current_pattern == 2 and p2_dip_ready) or (current_pattern == 3 and p3_dip_ready)
        if macd_golden_cross and strategy.position_size == 0 and na(signal_long)
            signal_long := 'MACD'

    // 3. 试探做空 (P4)
    if current_pattern == 4 and strategy.position_size == 0
        if s1_short_ok
            signal_short := 'S1'
        else if s2_breakout_short
            signal_short := 'S2'

// ==================== OBV过滤器执行 ====================
if not obv_allow_long
    signal_long := na
if not obv_allow_short
    signal_short := na

// ==================== 出场条件 ====================
// 牛市P1形态切换平多：持多到达P1后，形态再切换则平多
if strategy.position_size > 0 and current_pattern == 1
    has_reached_p1 := true

// 熊市P4形态切换平空：持空到达P4后，形态再切换则平空
if strategy.position_size < 0 and current_pattern == 4
    has_reached_p4 := true

bool exit_long_pattern = enable_p1_switch_exit and strategy.position_size > 0 and has_reached_p1 and current_pattern != 1
bool exit_short_pattern = enable_p4_switch_exit and strategy.position_size < 0 and has_reached_p4 and current_pattern != 4

// ==================== 做多入场 ====================
if not na(signal_long)
    cycle_start_equity := strategy.equity
    entry_pattern_num := current_pattern
    entry_atr := atr_value
    has_reached_p1 := (current_pattern == 1)  // 修复：进场即P1则立刻激活，防止形态秒切导致退出逻辑失效
    p2_dip_ready := false
    p3_dip_ready := false

    entry_system := signal_long
    display_system := signal_long

    [current_risk, current_mult] = get_risk_params(current_pattern, true)

    // P1冷却期过滤（仅对P1生效）
    if current_pattern == 1 and enable_p1_cooldown and not na(p1_loss_bar) and (bar_index - p1_loss_bar) <= p1_cooldown_bars
        current_mult := p1_cooldown_mult

    entry_risk_percent    := current_risk
    entry_size_multiplier := current_mult

    float qty = calculate_position_size(close, atr_value, atr_mult_init, current_risk, current_mult)

    if qty > 0
        string entry_id = 'Pattern' + str.tostring(current_pattern) + 'Long'
        string comment = display_system + ' Long-P' + str.tostring(current_pattern)
        strategy.entry(entry_id, strategy.long, qty=qty, comment=comment)
        pyramid_count_long := 0
        entry_price_first := close

// ==================== 做空入场 ====================
if not na(signal_short)
    cycle_start_equity := strategy.equity
    entry_pattern_num := current_pattern
    entry_atr := atr_value
    has_reached_p4 := (current_pattern == 4)  // 修复：进场即P4则立刻激活，防止形态秒切导致退出逻辑失效
    p5_breakout_ready := false
    p6_breakout_ready := false

    // 【修复】直接读取授权系统，与做多入场对称
    entry_system := signal_short
    display_system := signal_short

    // 设置牛熊试探标记
    if enable_regime_filter
        if bull_market_active and current_pattern == 4
            regime_test_type := 'P4_bear_test'

    [current_risk, current_mult] = get_risk_params(current_pattern, false)
    entry_risk_percent := current_risk
    entry_size_multiplier := current_mult

    float qty = calculate_position_size(close, atr_value, atr_mult_init, current_risk, current_mult)

    if qty > 0
        string entry_id = 'Pattern' + str.tostring(current_pattern) + 'Short'
        string comment = display_system + ' Short-P' + str.tostring(current_pattern)
        strategy.entry(entry_id, strategy.short, qty=qty, comment=comment)
        pyramid_count_short := 0
        entry_price_first := close

// ==================== 出场执行 ====================
if exit_long_pattern
    strategy.close_all(comment='P1 Pattern Switch Exit Long')

if exit_short_pattern
    strategy.close_all(comment='P4 Pattern Switch Exit Short')

// ==================== 多头ATR止损 ====================
if strategy.position_size > 0
    if strategy.position_size[1] <= 0
        long_stop_level := strategy.position_avg_price - entry_atr * atr_mult_init
        long_trail_active := false
        long_trail_just_activated := false
    else
        float current_profit = high - strategy.position_avg_price
        if enable_trailing_stop
            if not long_trail_active and current_profit >= atr_value * profit_atr_mult_long
                long_trail_active := true
                long_trail_just_activated := true
                float calculated_trail_long = high - atr_value * atr_mult_long_trail
                long_stop_level := use_breakeven ? math.max(strategy.position_avg_price, calculated_trail_long) : calculated_trail_long
            if long_trail_active and not long_trail_just_activated
                float new_stop = close - atr_value * atr_mult_long_trail
                long_stop_level := math.max(long_stop_level, new_stop)

    if not na(long_stop_level)
        if not long_trail_active
            if close < long_stop_level
                strategy.close_all(comment='Initial SL')
        else if not long_trail_just_activated
            if close < long_stop_level
                strategy.close_all(comment='Trailing SL')

// ==================== 空头ATR止损 ====================
if strategy.position_size < 0
    if strategy.position_size[1] >= 0
        short_stop_level := strategy.position_avg_price + entry_atr * atr_mult_init
        short_trail_active := false
        short_trail_just_activated := false
    else
        float current_profit = strategy.position_avg_price - low
        if enable_trailing_stop
            if not short_trail_active and current_profit >= atr_value * profit_atr_mult_short
                short_trail_active := true
                short_trail_just_activated := true
                float calculated_trail_short = low + atr_value * atr_mult_short_trail
                short_stop_level := use_breakeven ? math.min(strategy.position_avg_price, calculated_trail_short) : calculated_trail_short
            if short_trail_active and not short_trail_just_activated
                float new_stop = close + atr_value * atr_mult_short_trail
                short_stop_level := math.min(short_stop_level, new_stop)

    if not na(short_stop_level)
        if not short_trail_active
            if close > short_stop_level
                strategy.close_all(comment='Initial SL')
        else if not short_trail_just_activated
            if close > short_stop_level
                strategy.close_all(comment='Trailing SL')

// ==================== 多头加仓（追踪止损激活时加仓，支持0/1/2次） ====================
if strategy.position_size > 0 and pyramid_enable and pyramid_max_count_long > 0 and pyramid_count_long < pyramid_max_count_long
    if long_trail_just_activated
        pyramid_count_long := pyramid_count_long + 1
        float base_qty = calculate_position_size(close, atr_value, atr_mult_init, entry_risk_percent, entry_size_multiplier)
        float add_qty = math.round(math.floor(base_qty * pyramid_mult_u2 / qty_step + 1e-9) * qty_step, 6)
        if add_qty >= min_qty
            string add_comment = 'Scale-in Long ' + str.tostring(pyramid_count_long)
            strategy.entry('Scale-in Long' + str.tostring(pyramid_count_long), strategy.long, qty=add_qty, comment=add_comment)

// 多头加仓后下一根K线更新止损（跟随use_breakeven开关）
if strategy.position_size > 0 and pyramid_count_long > 0 and long_trail_active and not long_trail_just_activated
    if use_breakeven
        long_stop_level := math.max(long_stop_level, strategy.position_avg_price)
    else
        float new_stop = math.max(close - atr_value * atr_mult_long_trail, entry_price_first)
        long_stop_level := math.max(long_stop_level, new_stop)

// ==================== 空头加仓（追踪止损激活时加仓，支持0/1/2次） ====================
if strategy.position_size < 0 and pyramid_enable and pyramid_max_count_short > 0 and pyramid_count_short < pyramid_max_count_short
    if short_trail_just_activated
        pyramid_count_short := pyramid_count_short + 1
        float base_qty_s = calculate_position_size(close, atr_value, atr_mult_init, entry_risk_percent, entry_size_multiplier)
        float add_qty_s = math.round(math.floor(base_qty_s * pyramid_mult_u2 / qty_step + 1e-9) * qty_step, 6)
        if add_qty_s >= min_qty
            string add_comment_s = 'Scale-in Short ' + str.tostring(pyramid_count_short)
            strategy.entry('Scale-in Short' + str.tostring(pyramid_count_short), strategy.short, qty=add_qty_s, comment=add_comment_s)

// 空头加仓后下一根K线更新止损（跟随use_breakeven开关）
if strategy.position_size < 0 and pyramid_count_short > 0 and short_trail_active and not short_trail_just_activated
    if use_breakeven
        short_stop_level := math.min(short_stop_level, strategy.position_avg_price)
    else
        float new_stop = math.min(close + atr_value * atr_mult_short_trail, entry_price_first)
        short_stop_level := math.min(short_stop_level, new_stop)

// 重置追踪止损激活标志
if long_trail_just_activated
    long_trail_just_activated := false
if short_trail_just_activated
    short_trail_just_activated := false

// ==================== 实时止损执行(放在主循环) ====================
if enable_realtime_stop
    if strategy.position_size > 0
        float realtime_stop_price_long = strategy.position_avg_price * (1 - realtime_stop_percent / 100)
        // 【修复】去掉 from_entry，让止损对底仓和所有加仓单同时生效
        strategy.exit('Price SL Long', stop=realtime_stop_price_long, comment='Price SL')
    if strategy.position_size < 0
        float realtime_stop_price_short = strategy.position_avg_price * (1 + realtime_stop_percent / 100)
        // 【修复】去掉 from_entry，让止损对底仓和所有加仓单同时生效
        strategy.exit('Price SL Short', stop=realtime_stop_price_short, comment='Price SL')

// ==================== 实时止损 display ====================
var float realtime_stop_display = na
if strategy.position_size > 0
    realtime_stop_display := enable_realtime_stop ? strategy.position_avg_price * (1 - realtime_stop_percent / 100) : na
else if strategy.position_size < 0
    realtime_stop_display := enable_realtime_stop ? strategy.position_avg_price * (1 + realtime_stop_percent / 100) : na
else
    realtime_stop_display := na

// ==================== 图表绘制 ====================
plot(strategy.position_avg_price, '入场均价', color=color.new(color.gray, 44), linewidth=1, style=plot.style_linebr)
plot(strategy.position_size > 0 ? long_stop_level : na, '多头ATR止损', color=color.new(color.fuchsia, 44), linewidth=1, style=plot.style_linebr)
plot(strategy.position_size < 0 ? short_stop_level : na, '空头ATR止损', color=color.new(color.fuchsia, 44), linewidth=1, style=plot.style_linebr)
plot(realtime_stop_display, '实时止损线', color=color.new(color.maroon, 44), linewidth=1, style=plot.style_linebr)

// 双系统通道线
plot(use_dual_system ? s1_high_long : na, 'S1多头上轨', color=color.new(color.lime, 44), linewidth=1)
plot(use_dual_system ? s1_low_short : na, 'S1空头下轨', color=color.new(color.red, 44), linewidth=1)
plot(s2_high_long, 'S2上轨', color=color.new(color.lime, 11), linewidth=2)
plot(s2_low_short, 'S2下轨', color=color.new(color.red, 11), linewidth=2)


// 牛熊状态背景色
bgcolor(enable_regime_filter ? (bull_market_active ? color.new(color.green, 92) : bear_market_active ? color.new(color.red, 92) : color.new(color.gray, 95)) : color.new(color.green, 92), title='牛熊状态')


// 形态标签
bool pattern_changed = current_pattern != current_pattern[1]
plotshape(pattern_changed[1] and current_pattern[1] == 1, style=shape.labeldown, location=location.abovebar, color=color.green,  text='1', textcolor=color.white, size=size.small)
plotshape(pattern_changed[1] and current_pattern[1] == 2, style=shape.labeldown, location=location.abovebar, color=color.aqua,   text='2', textcolor=color.white, size=size.small)
plotshape(pattern_changed[1] and current_pattern[1] == 3, style=shape.labeldown, location=location.abovebar, color=color.maroon, text='3', textcolor=color.white, size=size.small)
plotshape(pattern_changed[1] and current_pattern[1] == 4, style=shape.labeldown, location=location.abovebar, color=color.red,    text='4', textcolor=color.white, size=size.small)
plotshape(pattern_changed[1] and current_pattern[1] == 5, style=shape.labeldown, location=location.abovebar, color=color.orange, text='5', textcolor=color.white, size=size.small)
plotshape(pattern_changed[1] and current_pattern[1] == 6, style=shape.labeldown, location=location.abovebar, color=color.yellow, text='6', textcolor=color.white, size=size.small)

// ==================== 持仓信息标签 ====================
var label position_label = na

if strategy.position_size != 0
    // 1. 拼接文本
    float profit_pct = strategy.position_size > 0 ? (close - strategy.position_avg_price) / strategy.position_avg_price * 100 : (strategy.position_avg_price - close) / strategy.position_avg_price * 100

    string labelText = 'P' + str.tostring(entry_pattern_num) + (strategy.position_size > 0 ? 'Long' : 'Short')
    if not na(display_system)
        labelText := labelText + '[' + display_system + ']'
    if s1_last_failed
        labelText := labelText + ' ⚠️S1 Disabled'
    if enable_obv_filter
        labelText := labelText + (obv_osc > 0 ? ' OBV↑' : obv_osc < 0 ? ' OBV↓' : ' OBV=')
    if enable_regime_filter
        labelText := labelText + (bull_market_active ? ' 🐮' : bear_market_active ? ' 🐻' : '')
        if not na(regime_test_type)
            labelText := labelText + ' 🔄Trial'

    labelText := labelText + '\n盈亏: ' + str.tostring(profit_pct, '#.2') + '%'

    float current_sl = strategy.position_size > 0 ? long_stop_level : short_stop_level
    if not na(current_sl)
        labelText := labelText + '\nATR SL: ' + str.tostring(current_sl, '#.2')

    if enable_realtime_stop
        float rt_sl = strategy.position_avg_price * (1 + (strategy.position_size > 0 ? -1 : 1) * realtime_stop_percent / 100)
        labelText := labelText + '\nPrice SL: ' + str.tostring(rt_sl, '#.2')

    int cur_pyramid_max = strategy.position_size > 0 ? pyramid_max_count_long : pyramid_max_count_short
    int cur_pyramid = strategy.position_size > 0 ? pyramid_count_long : pyramid_count_short
    if pyramid_enable
        if cur_pyramid_max > 0
            labelText := labelText + '\nScale-in: ' + str.tostring(cur_pyramid) + '/' + str.tostring(cur_pyramid_max) + (cur_pyramid >= cur_pyramid_max ? ' Done' : ' Pending')
    else
        labelText := labelText + '\nScale-in: Disabled'

    // 2. 设定样式
    color labelColor = strategy.position_size > 0 ? color.green : color.red
    string labelStyle = strategy.position_size > 0 ? label.style_label_down : label.style_label_up
    float yPos = strategy.position_size > 0 ? high * 1.003 : low * 0.997

    // 3. 【优化】高性能渲染：不存在则新建，存在则只更新，避免每K线创建销毁的性能损耗
    if na(position_label)
        position_label := label.new(x=bar_index, y=yPos, text=labelText, color=labelColor, style=labelStyle, textcolor=color.white)
    else
        label.set_xy(position_label, bar_index, yPos)
        label.set_text(position_label, labelText)
        label.set_color(position_label, labelColor)
        label.set_style(position_label, labelStyle)
else
    // 无持仓时删除标签并置空
    if not na(position_label)
        label.delete(position_label)
        position_label := na
